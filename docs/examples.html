11 - Class with Private Fields

Class with private fields and nullable return types

11_class.bl 
import std::io

class SomeObject {
  some_int: int = 10
  some_float: float = 5.0
  some_string: str = "Default string"

  private some_private: bool = false

  fn someMethod(): str? {
    if (!some_private) {
      some_private = true
      return null
    }
    return some_string
  }
}

let some_object = SomeObject()

println("Call 1: ${some_object.someMethod()}")
println("Call 2: ${some_object.someMethod()}")

22 - HTTP Client

Full-featured HTTP client with enums and configuration

22_http_client.bl 
// Full-featured HTTP Client example for Bengal
// Demonstrates enums and HttpClient

import std::http

// Using HttpMethod enum
let method = HttpMethod.GET

// Using RedirectPolicy enum
let policy = RedirectPolicy.Follow

// Print enum values (they are integers)
println("HTTP Client example loaded!")
println("HttpMethod.GET value: ${method}")
println("RedirectPolicy.Follow value: ${policy}")

// Create HTTP client
let client = HttpClient.new()
client.set_timeout(5000)
client.set_verify_ssl(true)

// Add default headers
client.add_header("User-Agent", "Bengal/1.0")
client.add_header("Accept", "application/json")

println("HttpClient created and configured!")
println("Ready to make HTTP requests!")

26 - Reflection & JSON

Runtime type inspection and JSON serialization

26_reflection.bl 
// Test JSON and Reflection
import std::io

class User {
  name: str = "Alice"
  age: int = 30
}

let u = User()

println("Original object:")
println(u.name)

// Reflection
let type = std::reflect::type_of(u)
println("Type: ${type}")

let className = std::reflect::class_name(u)
println("Class: ${className}")

let fields = std::reflect::fields(u)
println("Fields map (as object):")
println("fields.name: ${fields.name}")

// JSON
let jsonStr = std::json::stringify(u)
println("JSON: ${jsonStr}")

let parsed = std::json::parse(jsonStr)
println("Parsed object class: ${std::reflect::class_name(parsed)}")
println("Parsed name: ${parsed.name}")
println("Parsed age: ${parsed.age}")

// Test primitives
let numJson = std::json::stringify(123)
println("Num JSON: ${numJson}")
let parsedNum = std::json::parse(numJson)
println("Parsed Num type: ${std::reflect::type_of(parsedNum)}")
